**自动生成CNI配置的脚本**

```bash
#!/bin/bash
INF=enp0s1
R=$(($RANDOM % 250))

mkdir -p /etc/cni/net.d/
cat > /etc/cni/net.d/10-mynet.conflist << EOF
{
   "name":"cni0",
   "cniVersion":"0.3.1",
   "plugins":[
      {
         "nodename":"$(hostname)",
         "name":"myipvlan",
         "type":"ipvlan",
         "master":"$INF",
         "mode":"l2",
         "ipam":{
            "type":"host-local",
            "subnet":"10.244.0.0/24",
            "rangeStart":"10.244.$R.100",
            "rangeEnd":"10.244.$R.230",
            "routes":[
               {
                  "dst":"0.0.0.0/0"
               }
            ]
         }
      }
   ]
}
EOF
```
